🧠 2. LLM CLIENT (SINGLE CALL = CHEAP)
llm_client.py

Use Gemini Flash or OpenAI mini model.

import requests

def generate_post(trend):
    prompt = f"""
    You are a LinkedIn expert content creator.

    Trend: {trend}

    Write:
    1. Viral LinkedIn post (max 1500 chars)
    2. 5-8 hashtags
    3. Image prompt for AI generation

    Format:
    POST:
    ...
    HASHTAGS:
    ...
    IMAGE_PROMPT:
    ...
    """

    response = requests.post(
        "YOUR_LLM_ENDPOINT",
        json={"prompt": prompt}
    )

    return response.json()["text"]

👉 This is your ONLY AI cost per post

🧠 3. CONTENT PARSER (SPLIT OUTPUT)
def parse_output(text):
    sections = text.split("HASHTAGS:")
    post = sections[0].replace("POST:", "").strip()

    hashtags = sections[1].split("IMAGE_PROMPT:")[0].strip()
    image_prompt = sections[1].split("IMAGE_PROMPT:")[1].strip()

    return post, hashtags, image_prompt

🔗 5. LINKEDIN PUBLISHER (OFFICIAL API)
linkedin_api.py
import requests

ACCESS_TOKEN = "YOUR_LINKEDIN_TOKEN"

def post_to_linkedin(text, image_url=None):
    url = "https://api.linkedin.com/v2/ugcPosts"

    headers = {
        "Authorization": f"Bearer {ACCESS_TOKEN}",
        "Content-Type": "application/json"
    }

    payload = {
        "author": "urn:li:person:YOUR_ID",
        "lifecycleState": "PUBLISHED",
        "specificContent": {
            "com.linkedin.ugc.ShareContent": {
                "shareCommentary": {"text": text},
                "shareMediaCategory": "NONE"
            }
        },
        "visibility": {
            "com.linkedin.ugc.MemberNetworkVisibility": "PUBLIC"
        }
    }

    requests.post(url, json=payload, headers=headers)
🧠 6. MAIN PIPELINE (CORE LOGIC)
main.py
from tools.trend_loader import get_trend
from tools.llm_client import generate_post
from agents.content_generator import parse_output
from publisher.linkedin_api import post_to_linkedin

def run():
    trend = get_trend()

    raw_output = generate_post(trend)

    post, hashtags, image_prompt = parse_output(raw_output)

    final_post = f"{post}\n\n{hashtags}"

    post_to_linkedin(final_post)

if __name__ == "__main__":
    run()
⏰ 7. SCHEDULER (AUTOMATION)
scheduler/daily_runner.py
import time
import schedule
from main import run

schedule.every().day.at("10:00").do(run)

while True:
    schedule.run_pending()
    time.sleep(60)
